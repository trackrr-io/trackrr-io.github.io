        final GpsClientManager client = new GpsClientManager(gpsKey, secret);
        client.connect(new AWSIotMqttClientStatusCallback() {
            @Override
            public void onStatusChanged(final AWSIotMqttClientStatus status, final Throwable throwable) {
                if (status == AWSIotMqttClientStatus.Connected) {
                    final Gps gps = new Gps();
                    final Map<String, String> userData = new HashMap<>();
                    gps.setGpsKey(gpsKey);
                    gps.setCreatedDate(System.currentTimeMillis());
                    gps.setUserData(userData);
                    final Coords coords = new Coords();
                    coords.setLatitude(33.79519271850586d);
                    coords.setLongitude(-118.01104736328125d);
                    gps.setCoords(coords);
                    client.publish(gps);
                }
            }
        });
